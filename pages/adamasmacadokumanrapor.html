Adam Asmaca oyununun js kodlarını adım adım açıklayacağım;
Öncelikle değişkenlerimizi tanımlıyoruz. 
    var kelimeler = [
    "istanbul",
    "ankara",
    "kayseri",
    "izmir",
    "manisa",
    "bursa",
    "antalya",
    "trabzon",
    "samsun",
    "muğla",
    "marmaris"
    ]

    let cevap = '';
    let tahmin = [];
    let hata = 0;
    let maxYanlış = 6;
    let durum = null;

Oyuna başlama ekranındayken reset butonunun gözükmesi manasız olur. O yüzden oyun başlama ekranında reset butonunu gizledim. 
    document.getElementById('btnReset').style.visibility='hidden';

Oyundaki butonların js kodlarını şu şekilde tasarladım; 
    function butonlar() {
        let btnHTML = 'abcçdefgğhiıjklmnoöprsştuüvyz'.split('').map(harf =>
            `
          < button
                  class="harfbuton"
                  id='` + harf + `'
                  onClick="tahminFonk('` + harf + `')"
          >
            ` + harf + `
          < /button>
        `).join('');

        document.getElementById('klavye').innerHTML = btnHTML;
    }

Kelime havuzumdan random bir kelime getirmek için Math.floor ve Math.random fonksiyonlarını kullandım. 
    function randomKelime() {
        cevap = kelimeler[Math.floor(Math.random() * kelimeler.length)];
    }

Girilen tahmini kelimenin doğru veya yanlış olduğunu kontrol eden fonksiyon şu şekildedir; 
    function tahminFonk(seçilenKelime) {
        tahmin.indexOf(seçilenKelime) === -1 ? tahmin.push(seçilenKelime) : null;
        document.getElementById(seçilenKelime).setAttribute('disabled', true);

        if (cevap.indexOf(seçilenKelime) >= 0) {
            tahminKelime();
            kazanmaKontrol();
        } else if (cevap.indexOf(seçilenKelime) === -1) {
            hata++;
            hataGüncelleme();
            kaybetmeKontrol();
            resmiGüncelle();
        }
    }

Kazanma durumunu kontrol eder. Kelimedeki tüm harfler doğru şekilde seçildiyse oyun kazanılır. 
    function kazanmaKontrol() {
        if (durum === cevap) {
            document.getElementById('klavye').innerHTML = 'Tebrikler Kazandınız!';
        }
    }

Oyunda hata yaptıkça adamın asılması evresinin ilerlemesi gerek. Bu yüzden her hatada resim güncellenir. (Çizdirerek yapmadım resim bastım.) 
    function resmiGüncelle() {
        document.getElementById('adamAsmaca Foto').src = './images/' + hata + '.jpg';
    }

Oyunda hata yapıldıkça sayaç olarak kullandığım hata değişkeni güncellenir. 
    function hataGüncelleme() {
        document.getElementById('hata').innerHTML = hata;
    }

Kaybetme durumunu kontrol eder. Maksimum Yanlışa (6) erişildiği an oyun sona erer ve doğru kelime ekrana gelir. 
    function kaybetmeKontrol() {
        if (hata === maxYanlış) {
         document.getElementById('kelimeleriGoster').innerHTML = 'Cevap bu idi: ' + cevap;
            document.getElementById('klavye').innerHTML = 'Kaybettiniz!';
        }
    }

Tahmin edilen kelimeyi kontrol eder. 
    function tahminKelime() {
        durum = cevap.split('').map(harf => (tahmin.indexOf(harf) >= 0 ? harf : " _ ")).join('');
        document.getElementById('kelimeleriGoster').innerHTML = durum;
    }

Oyuncu, oyunun ortasında yeni oyuna başlamak istediğinde aşağıdaki senaryolar gerçekleşir. (Yeni Oyun Butonu ile ilişkilendirildi.) 
    function sıfırla() {
        hata = 0;
        tahmin = [];
        randomKelime();
        tahminKelime();
        hataGüncelleme();
        butonlar();
        document.getElementById('adamAsmaca Foto').src = './images/' + 0 + '.jpg';
        document.getElementById('btnBasla').style.visibility='hidden';
    }

Oyuncu, oyuna başlamak istediğinde aşağıdaki senaryolar gerçekleşir. Başla butonu gizlenir. Yeni Oyun Butonu ise görünür olur. 
    function basla(){
        randomKelime();
        butonlar();
        tahminKelime();
        hataGüncelleme();
        document.getElementById('btnBasla').style.visibility='hidden';
        document.getElementById('btnReset').style.visibility='visible';
        document.getElementById('btnReset').style.marginRight='8%';
    }
